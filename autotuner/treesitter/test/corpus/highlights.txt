autotuner Branchy (int x) -> (int y) {
  start -> check;
  state check {
    if (x > 0) {
      -> positive;
    } elif (x == 0) {
      -> zero;
    } else {
      -> negative;
    }
  }
  state positive { terminal; }
  state zero     { terminal; }
  state negative { terminal; }
}

autotuner ErrorViewer -> (bool outer_completed, bool inner_completed) {
  uses DoNothing, FatalOtherError;
  outer_completed = false;
  inner_completed = false;
  start -> init;
  state init {
      outer_completed = DoNothing();
      -> done;
  }
  state done {
      Error err = nil;
      inner_completed, err = FatalOtherError();
      -> exit;
  }
  state exit {
      terminal;
  }
}
autotuner DoNothing -> (bool completed) {
  completed = true;
  start -> init;
  state init {
      terminal; 
  }
}
autotuner FatalOtherError() -> (bool completed, Error err) {
  completed = false;
  err = nil;
  start -> init;
  state init {
      err = fatalErrorMsg("We are really broken :(");
      terminal;
  }
}
