autotuner Calculator (int a, int b) -> (int sum) {
  sum = 0;
  start -> calculate;

  state calculate {
    sum = a + b;
    -> done;
  }

  state done {
    terminal;
  }
}

autotuner Branchy (int x) -> (int y) {
  start -> check;
  state check {
    if (x > 0) {
      -> positive;
    } elif (x == 0) {
      -> zero;
    } else {
      -> negative;
    }
  }
  state positive { terminal; }
  state zero     { terminal; }
  state negative { terminal; }
}
